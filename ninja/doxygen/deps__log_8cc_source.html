00099       if (!RecordId(nodes[i]))
00100         return false;
00101       made_change = true;

00103   }

00106   if (!made_change) {

00114         if (deps->nodes[i] != nodes[i]) {
00115           made_change = true;
00116           break;

00118       }
00119     }
00120   }

00123   if (!made_change)

00303 bool DepsLog::Recompact(const string& path, string* err) {
00304   METRIC_RECORD(".ninja_deps recompact");
00305   printf("Recompacting deps...\n");
00306 
00307   Close();

00309 
00310   // OpenForWrite() opens for append.  Make sure it's not appending to a
00311   // left-over file from a previous recompaction attempt that crashed somehow.
00312   unlink(temp_path.c_str());
00313

00315   if (!new_log.OpenForWrite(temp_path, err))
00316     return false;
00317 
00318   // Clear all known ids so that new ones can be reassigned.  The new indices
00319   // will refer to the ordering in new_log, not in the current log.
00320   for (vector<Node*>::iterator i = nodes_.begin(); i != nodes_.end(); ++i)
00321     (*i)->set_id(-1);
00322   
00323   // Write out all deps again.
00324   for (int old_id = 0; old_id < (int)deps_.size(); ++old_id) {
00325     Deps* deps = deps_[old_id];
00326     if (!deps) continue;  // If nodes_[old_id] is a leaf, it has no deps.
00327 
00328     if (!IsDepsEntryLiveFor(nodes_[old_id]))
00329       continue;
00330 
00331     if (!new_log.RecordDeps(nodes_[old_id], deps->mtime,
00332                             deps->node_count, deps->nodes)) {
00333       new_log.Close();
00334       return false;
00335     }
00336   }
00337 
00338   new_log.Close();

00340   // All nodes now have ids that refer to new_log, so steal its data.
00341   deps_.swap(new_log.deps_);

00343 
00344   if (unlink(path.c_str()) < 0) {
00345     *err = strerror(errno);
00346     return false;
00347   }
00348 
00349   if (rename(temp_path.c_str(), path.c_str()) < 0) {
00350     *err = strerror(errno);

00357 bool DepsLog::IsDepsEntryLiveFor(Node* node) {
00358   // Skip entries that don't have in-edges or whose edges don't have a
00359   // "deps" attribute. They were in the deps log from previous builds, but
00360   // the the files they were for were removed from the build and their deps
00361   // entries are no longer needed.
00362   // (Without the check for "deps", a chain of two or more nodes that each
00363   // had deps wouldn't be collected in a single recompaction.)
00364   return node->in_edge() && !node->in_edge()->GetBinding("deps").empty();
