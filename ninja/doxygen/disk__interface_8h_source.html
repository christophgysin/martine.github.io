00024 /// Interface for accessing the disk.
00025 ///
00026 /// Abstract so it can be mocked out for tests.  The real implementation
00027 /// is RealDiskInterface.
00028 struct DiskInterface {
00029   virtual ~DiskInterface() {}
00030 
00031   /// stat() a file, returning the mtime, or 0 if missing and -1 on
00032   /// other errors.
00033   virtual TimeStamp Stat(const string& path) const = 0;
00034 
00035   /// Create a directory, returning false on failure.
00036   virtual bool MakeDir(const string& path) = 0;
00037 
00038   /// Create a file, with the specified name and contents
00039   /// Returns true on success, false on failure
00040   virtual bool WriteFile(const string& path, const string& contents) = 0;
00041 
00042   /// Read a file to a string.  Fill in |err| on error.
00043   virtual string ReadFile(const string& path, string* err) = 0;
00044 
00045   /// Remove the file named @a path. It behaves like 'rm -f path' so no errors
00046   /// are reported if it does not exists.
00047   /// @returns 0 if the file has been removed,
00048   ///          1 if the file does not exist, and
00049   ///          -1 if an error occurs.
00050   virtual int RemoveFile(const string& path) = 0;
00051 
00052   /// Create all the parent directories for path; like mkdir -p
00053   /// `basename path`.
00054   bool MakeDirs(const string& path);
00055 };
