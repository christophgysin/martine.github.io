00022 // MurmurHash2, by Austin Appleby
00023 static inline
00024 unsigned int MurmurHash2(const void* key, size_t len) {
00025   static const unsigned int seed = 0xDECAFBAD;
00026   const unsigned int m = 0x5bd1e995;
00027   const int r = 24;
00028   unsigned int h = seed ^ len;
00029   const unsigned char* data = (const unsigned char*)key;
00030   while (len >= 4) {
00031     unsigned int k;
00032     memcpy(&k, data, sizeof k);
00033     k *= m;
00034     k ^= k >> r;
00035     k *= m;
00036     h *= m;
00037     h ^= k;
00038     data += 4;
00039     len -= 4;
00040   }
00041   switch (len) {
00042   case 3: h ^= data[2] << 16;
00043   case 2: h ^= data[1] << 8;
00044   case 1: h ^= data[0];
00045     h *= m;
00046   };
00047   h ^= h >> 13;
00048   h *= m;
00049   h ^= h >> 15;
00050   return h;
00051 }
