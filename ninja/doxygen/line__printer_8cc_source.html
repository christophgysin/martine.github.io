00045 void LinePrinter::Print(string to_print, LineType type) {
00046   if (console_locked_) {
00047     line_buffer_ = to_print;
00048     line_type_ = type;
00049     return;
00050   }
00051 
00052 #ifdef _WIN32
00053   CONSOLE_SCREEN_BUFFER_INFO csbi;
00054   GetConsoleScreenBufferInfo(console_, &csbi);
00055 #endif
00056 
00057   if (smart_terminal_) {
00058 #ifndef _WIN32
00059     printf("\r");  // Print over previous line, if any.
00060 #else
00061     csbi.dwCursorPosition.X = 0;
00062     SetConsoleCursorPosition(console_, csbi.dwCursorPosition);
00063 #endif
00064   }
00065 
00066   if (smart_terminal_ && type == ELIDE) {
00067 #ifdef _WIN32
00068     // Don't use the full width or console will move to next line.
00069     size_t width = static_cast<size_t>(csbi.dwSize.X) - 1;
00070     to_print = ElideMiddle(to_print, width);
00071     // We don't want to have the cursor spamming back and forth, so
00072     // use WriteConsoleOutput instead which updates the contents of
00073     // the buffer, but doesn't move the cursor position.
00074     GetConsoleScreenBufferInfo(console_, &csbi);
00075     COORD buf_size = { csbi.dwSize.X, 1 };
00076     COORD zero_zero = { 0, 0 };
00077     SMALL_RECT target = {
00078       csbi.dwCursorPosition.X, csbi.dwCursorPosition.Y,
00079       static_cast<SHORT>(csbi.dwCursorPosition.X + csbi.dwSize.X - 1),
00080       csbi.dwCursorPosition.Y
00081     };
00082     CHAR_INFO* char_data = new CHAR_INFO[csbi.dwSize.X];
00083     memset(char_data, 0, sizeof(CHAR_INFO) * csbi.dwSize.X);
00084     for (int i = 0; i < csbi.dwSize.X; ++i) {
00085       char_data[i].Char.AsciiChar = ' ';
00086       char_data[i].Attributes = csbi.wAttributes;
00087     }
00088     for (size_t i = 0; i < to_print.size(); ++i)
00089       char_data[i].Char.AsciiChar = to_print[i];
00090     WriteConsoleOutput(console_, char_data, buf_size, zero_zero, &target);
00091     delete[] char_data;
00092 #else
00093     // Limit output to width of the terminal if provided so we don't cause
00094     // line-wrapping.
00095     winsize size;
00096     if ((ioctl(0, TIOCGWINSZ, &size) == 0) && size.ws_col) {
00097       to_print = ElideMiddle(to_print, size.ws_col);
00098     }
00099     printf("%s", to_print.c_str());
00100     printf("\x1B[K");  // Clear to end of line.
00101     fflush(stdout);
00102 #endif
00103 
00104     have_blank_line_ = false;
00105   } else {
00106     printf("%s\n", to_print.c_str());
00107   }
00108 }

00110 void LinePrinter::PrintOrBuffer(const char* data, size_t size) {

00120 void LinePrinter::PrintOnNewLine(const string& to_print) {
00121   if (console_locked_ && !line_buffer_.empty()) {
00122     output_buffer_.append(line_buffer_);
00123     output_buffer_.append(1, '\n');
00124     line_buffer_.clear();
00125   }
00126   if (!have_blank_line_) {
00127     PrintOrBuffer("\n", 1);
00128   }
00129   if (!to_print.empty()) {
00130     PrintOrBuffer(&to_print[0], to_print.size());
00131   }
00132   have_blank_line_ = to_print.empty() || *to_print.rbegin() == '\n';
00133 }
