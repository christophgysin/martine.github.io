00056 int LoadManifests(bool measure_command_evaluation) {
00057   string err;
00058   RealFileReader file_reader;
00059   State state;
00060   ManifestParser parser(&state, &file_reader);
00061   if (!parser.Load("build.ninja", &err)) {
00062     fprintf(stderr, "Failed to read test data: %s\n", err.c_str());
00063     exit(1);
00064   }
00065   // Doing an empty build involves reading the manifest and evaluating all
00066   // commands required for the requested targets. So include command
00067   // evaluation in the perftest by default.
00068   int optimization_guard = 0;
00069   if (measure_command_evaluation)
00070     for (size_t i = 0; i < state.edges_.size(); ++i)
00071       optimization_guard += state.edges_[i]->EvaluateCommand().size();
00072   return optimization_guard;
00073 }
