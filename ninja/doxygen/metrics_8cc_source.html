00061 int64_t TimerToMicros(int64_t dt) {
00062   static int64_t ticks_per_sec = 0;
00063   if (!ticks_per_sec) {
00064     LARGE_INTEGER freq;
00065     if (!QueryPerformanceFrequency(&freq))
00066       Fatal("QueryPerformanceFrequency: %s", GetLastErrorString().c_str());
00067     ticks_per_sec = LargeIntegerToInt64(freq);
00068   }
00069 
00070   // dt is in ticks.  We want microseconds.
00071   return (dt * 1000000) / ticks_per_sec;
00072 }
