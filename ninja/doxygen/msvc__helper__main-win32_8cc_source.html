00050 void WriteDepFileOrDie(const char* object_path, const CLParser& parse) {
00051   string depfile_path = string(object_path) + ".d";
00052   FILE* depfile = fopen(depfile_path.c_str(), "w");
00053   if (!depfile) {
00054     unlink(object_path);
00055     Fatal("opening %s: %s", depfile_path.c_str(),
00056           GetLastErrorString().c_str());
00057   }
00058   if (fprintf(depfile, "%s: ", object_path) < 0) {
00059     unlink(object_path);
00060     fclose(depfile);
00061     unlink(depfile_path.c_str());
00062     Fatal("writing %s", depfile_path.c_str());
00063   }
00064   const set<string>& headers = parse.includes_;
00065   for (set<string>::const_iterator i = headers.begin();
00066        i != headers.end(); ++i) {
00067     if (fprintf(depfile, "%s\n", EscapeForDepfile(*i).c_str()) < 0) {
00068       unlink(object_path);
00069       fclose(depfile);
00070       unlink(depfile_path.c_str());
00071       Fatal("writing %s", depfile_path.c_str());
00072     }
00073   }
00074   fclose(depfile);
00075 }
