00042     Fatal("pipe: %s", strerror(errno));

00046   // and so must avoid overly-large FDs.

00048     Fatal("pipe: %s", strerror(EMFILE));
00049 #endif  // !USE_PPOLL
00050   SetCloseOnExec(fd_);
00051 
00052   pid_ = fork();
00053   if (pid_ < 0)
00054     Fatal("fork: %s", strerror(errno));
00055 
00056   if (pid_ == 0) {
00057     close(output_pipe[0]);
00058 
00059     // Track which fd we use to report errors on.
00060     int error_pipe = output_pipe[1];
00061     do {
00062       if (sigaction(SIGINT, &set->old_act_, 0) < 0)
00063         break;
00064       if (sigprocmask(SIG_SETMASK, &set->old_mask_, 0) < 0)
00065         break;
00066 
00067       if (!use_console_) {
00068         // Put the child in its own process group, so ctrl-c won't reach it.
00069         if (setpgid(0, 0) < 0)
00070           break;
00071 
00072         // Open /dev/null over stdin.
00073         int devnull = open("/dev/null", O_RDONLY);
00074         if (devnull < 0)
00075           break;
00076         if (dup2(devnull, 0) < 0)
00077           break;
00078         close(devnull);
00079 
00080         if (dup2(output_pipe[1], 1) < 0 ||
00081             dup2(output_pipe[1], 2) < 0)
00082           break;
00083 
00084         // Now can use stderr for errors.
00085         error_pipe = 2;
00086         close(output_pipe[1]);
00087       }
00088       // In the console case, output_pipe is still inherited by the child and
00089       // closed when the subprocess finishes, which then notifies ninja.
00090 
00091       execl("/bin/sh", "/bin/sh", "-c", command.c_str(), (char *) NULL);
00092     } while (false);
00093 
00094     // If we get here, something went wrong; the execl should have
00095     // replaced us.
00096     char* err = strerror(errno);
00097     if (write(error_pipe, err, strlen(err)) < 0) {
00098       // If the write fails, there's nothing we can do.
00099       // But this block seems necessary to silence the warning.
00100     }
00101     _exit(1);
00102   }
00103 
00104   close(output_pipe[1]);
